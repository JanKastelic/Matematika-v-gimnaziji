    \subsection{Osnove logike}

        \begin{frame}
            \frametitle{Izjave}

            \only<2->{\begin{alertblock}{Matematična izjava}
                \only<3->{\textbf{Matematična izjava} je vsaka smiselna poved, za katero 
                lahko določimo resničnost oziroma pravilnost.}
            \end{alertblock}}

            \only<4->{\begin{alertblock}{Logična vrednost matematične izjave}
                \only<5->{Matematična izjava lahko zavzame dve logični vrednosti:
                \begin{itemize}
                    \item<6-> izjava je \textbf{resnična}/\textbf{pravilna}, 
                        oznaka $\mathbf{R}$/$\mathbf{P}$/$\mathbf{1}$/$\mathbf{\top}$;
                    \item<7-> izjava je \textbf{neresnična}/\textbf{nepravilna}, 
                        oznaka $\mathbf{N}$/$\mathbf{0}$/$\mathbf{\bot }$.
                \end{itemize}}                
            \end{alertblock}}

            \only<8->{\begin{alertblock}{}
                Izjave označujemo z velikimi tiskanimi črkami ($A$, $B$, $C$ ...).
            \end{alertblock}}
        \end{frame}

        \begin{frame}
            \only<2->{\begin{exampleblock}{Naloga}
                Ali so naslednje povedi izjave?
                \only<3->{\begin{itemize}   
                    \item Danes sije sonce.
                    \item Koliko je ura?
                    \item Piramida je geometrijski lik.
                    \item Daj mi jabolko.
                    \item Število $12$ deli število $3$.
                    \item Število $3$ deli število $10$.
                    \item Ali si pisal matematični test odlično?
                    \item Matematični test si pisal odlično.
                    \item Ali je $10~dl$ isto kot $1~l$?
                    \item Število $41$ je praštevilo.
                \end{itemize}}
            \end{exampleblock}}
        \end{frame}

        \begin{frame}
            \only<2->{\begin{exampleblock}{Naloga}
                Spodnjim izjavam določite logične vrednosti.
                \only<3->{\begin{itemize}   
                    \item $A$: Najvišja gora v Evropi je Mont Blanc.
                    \item $B$: Število je deljivo s $4$ natanko takrat, ko je vsota števk deljiva s $4$.
                    \item $C$: Ostanek pri deljenju s $4$ je lahko $1$, $2$ ali $3$.
                    \item $D$: Mesec februar ima 28 dni.
                    \item $E$: Vsa praštevila so liha števila.
                    \item $F$: Število $1$ je naravno število.
                    \item $G$: Praštevil je neskončno mnogo.
                \end{itemize}}
            \end{exampleblock}}
            
        \end{frame}

        \begin{frame}
            \only<2->{\begin{alertblock}{Enostavne in sestavjene izjave}
                \only<3->{Izjave delimo med:}
                \begin{itemize}
                    \item<4-> \textbf{elementarne}/\textbf{enostavne izjave} -- ne moremo 
                        jih razstaviti na bolj enostavne;
                    \item<5-> \textbf{sestavljene izjave} -- sestavljene iz elementarnih izjav, 
                        ki jih med seboj povezujejo \textbf{logične operacije} (imenovane 
                        tudi izjavne povezave oziroma logična vezja).
                \end{itemize}
            \end{alertblock}}

            \only<6->{\begin{alertblock}{}
                Vrednost sestavljene izjave izračunamo glede na vrednosti elementarnih 
                izjav in izjavnih povezav med njimi.
            \end{alertblock}}
            
            \only<7->{\begin{alertblock}{}
                Pravilnost sestavljenih izjav nazorno prikazujejo 
                \textbf{resničnostne}/\textbf{pravilnostne tabele}.
            \end{alertblock}}

        \end{frame}

        \begin{frame}
            \frametitle{Logične operacije}

            \only<2->{\begin{alertblock}{Negacija}
                \only<3->{\textbf{Negacija} izjave $A$ je izjava, ki \textbf{trdi nasprotno} 
                kot izjava $A$.}
                \only<4->{$$ \mathbf{\lnot A} \quad \quad \textmd{\textbf{Ni res}, da velja izjava A.}$$}
            \end{alertblock}}

            \begin{columns}[T]
                \column{0.65\textwidth} 
                    \only<5->{\begin{alertblock}{}
                        Če je izjava $A$ pravilna, je $\lnot A$ nepravilna in obratno: 
                        če je $\lnot A$ pravilna, je $A$ nepravilna.
                    \end{alertblock}}
                    \only<7->{\begin{alertblock}{}
                        Negacija negacije izjave je potrditev izjave. $\lnot(\lnot A)=A$
                    \end{alertblock}}

                \column{0.3\textwidth} 
                \only<6->{\begin{table}
                    \centering
                    \begin{tabular}{||c|c||} 
                    \hhline{|t:==:t|}
                    \rowcolor[rgb]{0.843,0.718,0.718} $A$ & $\lnot A$  \\ 
                    \hhline{|:==:|}
                    $P$                                   & $N$                       \\ 
                    \hline
                    $N$                                   & $P$                       \\
                    \hhline{|b:==:b|}
                    \end{tabular}                    
                \end{table}}                

            \end{columns}
        \end{frame}

        \begin{frame}
            \only<2->{\begin{alertblock}{Konjunkcija}
                \only<3->{\textbf{Konjunkcija} izjav $A$ in $B$ nastane tako, da povežemo izjavi $A$ in $B$ 
                z \textbf{in (hkrati)}.}
                \only<4->{$$ \mathbf{A\land B} \quad \quad \textmd{Velja izjava A \textbf{in (hkrati)} izjava B.}$$}
            \end{alertblock}}

            \begin{columns}[T]
                \column{0.6\textwidth} 
                \only<5->{\begin{alertblock}{}
                        Če sta izjavi $A$ in $B$ pravilni, je pravilna tudi njuna konjunkcija, 
                        če je pa ena od izjav nepravilna, je nepravilna tudi njuna konjunkcija.
                    \end{alertblock}}

                \only<3->{\begin{itemize}
                    \item Število $28$ je večratnik števila $3$ in večkratnik števila $8$.
                    \item Število $7$ je praštevilo in je deljivo s številom $1$.
                    \item Vsakemu celemu številu lahko pripišemo nasprotno število in obratno število.
                    \item Ostanki pri deljenju števila s $3$ so lahko $0$, $1$ ali $2$,

        \begin{frame}
            \only<2->{\begin{alertblock}{Disjunkcija}
                \only<3->{\textbf{Disjunkcija} izjav $A$ in $B$ nastane s povezavo \textbf{ali}.}
                \only<4->{$$ \mathbf{A\lor B} \quad \quad \textmd{Velja izjava A \textbf{ali} izjava B 
                (lahko tudi obe hkrati).}$$}
            \end{alertblock}}

            \begin{columns}[T]
                \column{0.62\textwidth} 
                    \only<5->{\begin{alertblock}{}
                        Disjunkcija je nepravilna, če sta nepravilni obe izjavi, ki jo sestavljata,
                        v preostalih treh primerih je pravilna.
                    \end{alertblock}}

                \column{0.35\textwidth} 
                \only<6->{\begin{table}

                        \hhline{|:===:|}

                        \hline

                        \hline
                        $N$ & $N$ & $N$                         \\
                        \hhline{|b:===:b|}
                        \end{tabular}

            \end{columns}

            \only<2->{\begin{exampleblock}{Naloga}
                Določite logično vrednost disjunkcijam.
                \only<3->{\begin{itemize}
                    \item Število $24$ je večratnik števila $3$ ali $8$.
                    \item Število $35$ ni večratnik števila $7$ ali $6$.
                    \item Število $5$ deli število $16$ ali $18$.
                    \item Ploščina kvadrata s stranico $a$ je $a^2$ ali obseg kvadrata je $4a$.
                    \item Ni res, da je vsota notranjih kotov trikotnika $160^\circ$, ali ni res,

                \end{itemize}}
            \end{exampleblock}}
        \end{frame}


        \begin{frame}
            \only<2->{\begin{block}{Komutativnost konjunkcije in disjunkcije}
                \only<3->{$$ A\land B = B\land A  \quad \quad \quad
                 A\lor B = B\lor A$$}
            \end{block}}      
            
            \only<4->{\begin{block}{Asociativnost konjunkcije in disjunkcije}
                \only<5->{$$(A\land B)\land C = A\land(B\land C) \quad \quad \quad
                (A\lor B)\lor C = A\lor (B\lor C) $$}
            \end{block}}

            \only<6->{\begin{block}{Distributivnostna zakona za konjunkcijo in disjunkcijo}
                \only<7->{$$(A\lor B)\land C = (A\land C)\lor(B\land C) \quad \quad \quad
                (A\land B)\lor C = (A\lor C)\land(B\lor C) $$}
            \end{block}}

            \only<8->{\begin{block}{De Morganova zakona}
                \begin{itemize}
                    \item<9-> negacija konjunkcije je disjunkcija negacij: 
                        $\lnot(A\land B)=\lnot A\lor\lnot B$
                    \item<10-> negacija disjunkcije je konjunkcija negacij: 
                        $\lnot(A\lor B)=\lnot A\land\lnot B$
                \end{itemize}                
            \end{block}}
        
        \end{frame}

        \begin{frame}
            \only<2->{\begin{exampleblock}{Naloga}
                Katere od spodnjih izjav so pravilne in katere nepravilne?
                \only<3->{\begin{itemize}
                    \item $(3\cdot 4 = 12)\land(12:4=3)$
                    \item $(a^3\cdot a^5=a^{15})\lor (a^3\cdot a^5=a^8)$
                    \item $(3|30)\land(3|26)$
                    \item $(3|30)\lor(3|26)$
                    \item $(2^3=9)\lor(3^2=9)$
                    \item $((-2)^2=4)\land\lnot(-2^2=4)$
                \end{itemize}}
            \end{exampleblock}}
        \end{frame}

        \begin{frame}
            \only<2->{\begin{alertblock}{Implikacija}
                \only<3->{\textbf{Implikacija} izjav $A$ in $B$ je sestavljena izjava, ki jo lahko beremo
                na različne načine.}
                \only<4->{$$ \mathbf{A\Rightarrow B} \quad \quad \textmd{\textbf{Če} velja izjava A, 
                \textbf{potem} velja izjava B. / \textbf{Iz} A \textbf{sledi} B.}$$}
                \only<5->{Izjava $A$ je \textbf{pogoj} ali \textbf{privzetek}, izjava $B$ pa 
                \textbf{(logična) posledica} izjave $A$.}
            \end{alertblock}}

            \begin{columns}[T]
                \column{0.62\textwidth} 
                \only<6->{\begin{alertblock}{}
                        Implikacija je nepravilna, ko je izjava $A$ pravilna, izjava $B$ pa

                    \end{alertblock}}

        \begin{frame}
            \only<2->{\begin{alertblock}{Ekvivalenca}
                \only<3->{\textbf{Ekvivalenca} izjavi $A$ in $B$ poveže s \textbf{če in samo če} oziroma
                \textbf{natanko tedaj, ko}.}
                \only<4->{\begin{align*} 
                    \mathbf{A\Leftrightarrow B} \quad \quad &\textmd{Izjava A velja, \textbf{če in
                    samo če} velja izjava B.} / \\
                        &\textmd{Izjava A velja \textbf{natanko tedaj, ko} velja izjava B.}
                \end{align*}}
            \end{alertblock}}


            \begin{columns}[T]
                \column{0.62\textwidth} 
                \only<5->{\begin{alertblock}{}
                        Ekvivalenca dveh izjav je pravilna, če imata obe izjavi enako vrednost 
                        (ali sta obe pravilni ali obe nepravilni), in nepravilna, če imata izjavi
                        različno vrednost.
                    \end{alertblock}}
                    \only<7->{\begin{block}{}
                        Ekvivalentni/enakovredni izjavi pomenita eno in isto, lahko ju nadomestimo 
                        drugo z drugo.
                    \end{block}}

                    \item Število je deljivo s $24$ natanko takrat, ko je deljivo s $4$ in $6$ hkrati.
                    \item Število je praštevilo natanko takrat, ko ima natanko dva delitelja.

        \begin{frame}
            \only<2->{\begin{block}{Vrstni red operacij}
                \only<3->{Kadar so izjave povezane z več izjavnimi povezavami, pri določanju logične 
                vrednosti upoštevamo oklepaje in naslednji \textbf{vrstni red} oziroma
                \textbf{prioriteto izjavnih povezav}:}
                \begin{enumerate}
                    \item<4-> negacija,
                    \item<5-> konjunkcija,
                    \item<6-> disjunkcija,
                    \item<7-> implikacija,
                    \item<8-> ekvivalenca.
                \end{enumerate}
            \end{block}}

            \only<9->{\begin{block}{}
                Če moramo zapored izvesti več enakih izjavnih povezav, velja pravilo združevanja 
                od leve proti desni.
            \end{block}}
        \end{frame}

        \begin{frame}
            \only<2->{\begin{exampleblock}{Naloga}
                V sestavljeni izjavi zapišite oklepaje, ki bodo predstavljali vrstni red operacij.
                Nato tvorite pravilnostno tabelo za sestavljeno izjavo glede na različne logične 
                vrednosti elementarnih izjav.
                \only<3->{\begin{itemize}
                    \item $A\lor B\Leftrightarrow \lnot A\Rightarrow \lnot B$
                    \item $A\lor \lnot A\Rightarrow\lnot B\land (\lnot A\Rightarrow B) $
                    \item $A\Rightarrow B\Leftrightarrow \lnot B\Rightarrow \lnot A $
                    \item $A\land \lnot B\Leftrightarrow A \Rightarrow B$
                    \item $C\Rightarrow A\lor \lnot B\Leftrightarrow \lnot A\land C $
                    \item $\lnot A\lor \lnot B\Leftrightarrow B \land (C\Leftrightarrow \lnot A) $
                \end{itemize}}
            \end{exampleblock}}
        \end{frame}

        \begin{frame}
            \only<2->{\begin{alertblock}{Tavtologija}
                \only<3->{\textbf{Tavtologija} ali \textbf{logično pravilna izjava} je sestavljena izjava, 
                ki je pri vseh naborih vrednosti elementarnih izjav, iz katerih je sestavjena, pravilna.}
            \end{alertblock}}

            \only<4->{\begin{alertblock}{Protislovje}
                \only<5->{\textbf{Protislovje} je sestavljena izjava, ki ni nikoli pravilna.}
            \end{alertblock}}

            \only<6->{\begin{block}{Kvantifikatorja}
                \begin{itemize}
                    \item<7-> $\forall$ (beri '(za) vsak') -- izjava velja za vsak element dane množice
                    \item<8-> $\exists$ (beri 'obstaja' ali 'eksistira') -- izjava je pravilna za vsaj en element dane množice
                \end{itemize}
            \end{block}}

        \end{frame}

        \begin{frame}
            \frametitle{Pomen izjav v matematiki}
            \only<2->{\begin{block}{}
                \textbf{Aksiomi} so najpreprostejše izjave, ki so očitno pravilne in zato njihove 
                pravilnosti ni treba dokazovati.
            \end{block}}

            \only<3->{\begin{block}{}
                \textbf{Izreki} ali \textbf{teoremi} so izjave, ki so pravilne, vendar pa njihova 
                pravilnost ni očitna. 
                Pravilnost izreka (teorema) moramo potrditi z dokazom, ki temelji na aksiomih in na 
                preprostejših že prej dokazanih izrekih.
            \end{block}}

            \only<4->{\begin{block}{}
                \textbf{Definicije} so izjave, s katerimi uvajamo nove pojme. Najpreprostejših pojmov 
                v matematiki ne opisujemo z definicijami (to so pojmi kot npr.: število, premica ipd.); 
                vsak nadaljnji pojem pa moramo definirati, zato da se nedvoumno ve, o čem govorimo.
            \end{block}}

        \end{frame}
        
    \subsection{Osnove teorije množic}

        \begin{frame}
            \frametitle{Množice}
            \only<2->{\begin{alertblock}{Množica}
                \only<3->{\textbf{Množica} je skupek elementov, ki imajo neko skupno lastnost.}
            \end{alertblock}}

            \only<4->{\begin{block}{}
                Množica je določena, če:
                \begin{itemize}
                    \item<5-> lahko naštejemo vse njene elemente ali
                    \item<6-> poznamo pravilo/skupno lastnost, ki pove, kateri elementi so v množici.
                \end{itemize}
            \end{block}}

            \only<7->{\begin{block}{}
                Označujemo jih z velikimi črkami ($\mathcal{A}, \mathcal{B}, \mathcal{C} \dots$ ali  
                $A, B, C \dots$ ). 
            \end{block}}

            \only<8->{\begin{alertblock}{Univerzalna množica}
                \only<9->{\textbf{Univerzalna množica} ali \textbf{univerzum} ($\mathcal{U}$) je množica 
                vseh elementov, ki v danem primeru nastopajo oziroma jih opazujemo.}
            \end{alertblock}}

        \end{frame}      

        \begin{frame}
            \only<2->{\begin{alertblock}{Element množice}
                \only<3->{\textbf{Element množice} je objekt v množici. }
            \end{alertblock}}

            \only<4->{\begin{block}{}
                Označujemo jih z malimi črkami ($a, b, c \dots$). 
            \end{block}} 

            \only<5->{\begin{block}{}
                Elemente množice zapisujemo v zavitem oklepaju (npr. $\mathcal{A}=\left\{a, b, c\right\}$).
            \end{block}}

            \only<6->{\begin{block}{}
                Element je lahko vsebovan v množici (npr. $a\in\mathcal{A}$) ali pa 
                v množici ni vsebovan (npr. $d\notin\mathcal{A}$).
            \end{block} }

            \only<7->{\begin{alertblock}{Prazna množica}
                \only<8->{\textbf{Prazna množica} ($\mathbf{\emptyset, \left\{\right\}}$) je množica, 
                ki ne vsebuje nobenega elementa.}
            \end{alertblock}}

        \end{frame}

        \begin{frame}
            \frametitle{Moč množice}

            \only<2->{\begin{alertblock}{Moč množice}
                \only<3->{Število elementov v množici predstavlja \textbf{moč množice}.} 
                \only<4->{Oznaka: $\mathbf{m(\mathcal{A})}$ ali $\mathbf{|\mathcal{A}|}$.}
            \end{alertblock}}

            \only<5->{\begin{alertblock}{}
                Množica je lahko:
                \begin{itemize}
                    \item<6-> \textbf{končna množica} -- vsebuje končno mnogo elementov: $\mathbf{m(\mathcal{A})=n}$;
                    \item<7-> \textbf{neskončna množica} -- vsebuje neskončno mnogo elementov: $\mathbf{m(\mathcal{A})=\infty}$.
                \end{itemize}
            \end{alertblock}}

            \only<8->{\begin{block}{}
                Če ima množica toliko elementov, kot jih ima množica naravnih števil, je ta števno 
                neskončna.
                \only<9->{Njeno moč pišemo kot: $m(\mathcal{A})=\aleph_0$.}
            \end{block}}

            \only<10->{\begin{alertblock}{}
                Za množici, ki imata isto moč, rečemo, da sta \textbf{ekvipolentni} oziroma \textbf{ekvipotentni}.
            \end{alertblock}}

        \end{frame}

                        $\mathcal{B}$ velja, da je tudi element množice $\mathcal{A}$.}

                \begin{itemize}
                    \item<6-> $\forall \mathcal{A}:\mathcal{A}\subseteq\mathcal{A}$ -- Vsaka množica je podmnožica
                        same sebe.
                    \item<7-> $\forall \mathcal{A}:\emptyset\subseteq\mathcal{A}$ -- Prazna množica je podmnožica
                        vsake množice.
                \end{itemize}
            \end{block}}

            \only<8->{\begin{block}{}
                Moč podmnožice $\mathcal{B}$ množice $\mathcal{A}$ je manjša ali enaka moči množice $\mathcal{A}$:
                \only<9->{$$\mathcal{B}\subseteq\mathcal{A}\Rightarrow m(\mathcal{B})\leq m(\mathcal{A})$$}
            \end{block}}

        \begin{frame}
            \only<2->{\begin{block}{}
                Množici $\mathcal{A}$ in $\mathcal{B}$ sta \textbf{enaki}, če imata iste elemente; 
                sta druga drugi podmnožici.
                \only<3->{$$\mathcal{A}=\mathcal{B}\Leftrightarrow(\mathcal{A}\subseteq\mathcal{B})\land(\mathcal{B}\subseteq\mathcal{A})$$}
            \end{block}}

            \only<4->{\begin{block}{}
                Podmnožica $\mathcal{B}$ množice $\mathcal{A}$, ki ni enaka množici $\mathcal{A}$, 
                je \textbf{prava podmnožica} množice $\mathcal{A}$.
            \end{block}}

            \only<5->{\begin{alertblock}{Potenčna množica}
                \only<6->{\textbf{Potenčna množica} množice $\mathcal{A}$ je množica vseh podmnožic množice $\mathcal{A}$.}
                
                \only<7->{Oznaka: $\mathbf{\mathcal{P}\mathcal{A}}$ / $\mathbf{\mathcal{P}(\mathcal{A})}$.}
                \only<8->{$$ \mathcal{PA}=\left\{\mathcal{X}; \mathcal{X}\subseteq\mathcal{A} \right\}$$}
            \end{alertblock}}

            \only<9->{\begin{block}{}
                $$ m(\mathcal{PA})=2^{m(\mathcal{A})}$$
                \only<10->{Potenčna množica ni nikoli prazna -- vsebuje vsaj prazno množico.}
            \end{block}}
        \end{frame}

    \begin{frame}

        \begin{block}{}
            Za množici $\mathcal{A}$ in $\mathcal{B}$ velja:
            $$m(\mathcal{A}\cup\mathcal{B})=m(\mathcal{A})+m(\mathcal{B})-m(\mathcal{A}\cap\mathcal{B}) $$
        \end{block}

        \begin{block}{}
            Množici, katerih presek je prazna množica, sta \textbf{disjunktni} množici.
            $$\mathcal{A}\cap\mathcal{B}=\emptyset\Rightarrow m(\mathcal{A}\cap\mathcal{B})=0 $$ 
            $$\mathcal{A}\cap\mathcal{B}=\emptyset\Rightarrow m(\mathcal{A}\cup\mathcal{B})=m(\mathcal{A})+m(\mathcal{B}) $$
        \end{block}

        \begin{columns}[T]
            \column{0.475\textwidth}
                \begin{block}{Komutativnost unije in preseka}
                    $$ \mathcal{A}\cup\mathcal{B}=\mathcal{B}\cup\mathcal{A} $$
                    $$ \mathcal{A}\cap\mathcal{B}=\mathcal{B}\cap\mathcal{A} $$
                \end{block}

            \column{0.475\textwidth}
                \begin{block}{Asociativnost unije in preseka}
                    $$ \left(\mathcal{A}\cup\mathcal{B}\right)\cup\mathcal{C}=\mathcal{A}\cup\left(\mathcal{B}\cup\mathcal{C}\right) $$
                    $$ \left(\mathcal{A}\cap\mathcal{B}\right)\cap\mathcal{C}=\mathcal{A}\cap\left(\mathcal{B}\cap\mathcal{C}\right) $$
                \end{block}

        \end{columns}

    \end{frame}

    \begin{frame}
        \begin{block}{Distribustvnostna zakona za unijo in presek}
            $$ \left(\mathcal{A}\cup\mathcal{B}\right)\cap\mathcal{C}=\left(\mathcal{A}\cap\mathcal{C}\right)\cup\left(\mathcal{B}\cap\mathcal{C}\right) $$
            $$ \left(\mathcal{A}\cap\mathcal{B}\right)\cup\mathcal{C}=\left(\mathcal{A}\cup\mathcal{C}\right)\cap\left(\mathcal{B}\cup\mathcal{C}\right) $$
        \end{block}

        \begin{block}{De Morganova zakona}
            Komplement preseka dveh množic je enak uniji komplementov obeh množic:
            $$\left(\mathcal{A}\cap\mathcal{B}\right)^\complement=\mathcal{A}^\complement\cup\mathcal{B}^\complement. $$
            Komplement unije dveh množic je enak preseku komplementov obeh množic:
            $$\left(\mathcal{A}\cup\mathcal{B}\right)^\complement=\mathcal{A}^\complement\cap\mathcal{B}^\complement. $$
        \end{block}
    \end{frame}

                    $$\mathcal{A}\neq\mathcal{B}\Rightarrow \mathcal{A}\times\mathcal{B}\neq\mathcal{B}\times\mathcal{A}$$
                \end{block}

                \begin{block}{}
                    $$m(\mathcal{A}\times\mathcal{B})=m(\mathcal{A})\cdot m(\mathcal{B}) $$
                \end{block}
