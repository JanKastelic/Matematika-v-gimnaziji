        \begin{frame}
            \begin{block}{Komutativnost konjunkcije in disjunkcije}
                $$ A\land B = B\land A  \quad \quad \quad
                 A\lor B = B\lor A$$
            \end{block}      
            
            \begin{block}{Asociativnost konjunkcije in disjunkcije}
                $$(A\land B)\land C = A\land(b\land C) \quad \quad \quad
                (A\lor B)\lor C = A\lor (B\lor C) $$
            \end{block}

            \begin{block}{Distributivnost zakona za konjunkcijo in disjunkcijo}
                $$(A\lor B)\land C = (A\land C)\lor(B\land C) \quad \quad \quad
                (A\land B)\lor C = (A\lor C)\land(B\lor C) $$
            \end{block}

            \begin{block}{De Morganova zakona}
                \begin{itemize}
                    \item negacija konjunkcije je disjunkcija negacij: 
                        $\lnot(A\land B)=\lnot A\lor\lnot B$
                    \item negacija disjunkcije je konjunkcija negacij: 
                        $\lnot(A\lor B)=\lnot A\land\lnot B$
                \end{itemize}                
            \end{block}
        
        \end{frame}

        \begin{frame}
            \begin{block}{Vrstni red operacij}
                Kadar so izjave povezane z veƒç izjavnimi povezavami, pri doloƒçanju logiƒçne 
                vrednosti upo≈°tevamo oklepaje in naslednji \textbf{vrstni red} oziroma

                \begin{enumerate}

                \end{enumerate}
            \end{block}
            \begin{block}{}

                Nato tvorite pravilnostno tabelo za sestavljeno izjavo glede na razliƒçne logiƒçne 
                vrednosti elementarnih izjav.
                \begin{itemize}
                    \item $A\lor B\Leftrightarrow \lnot A\Rightarrow \lnot B$

                    \item $A\Rightarrow B\Leftrightarrow \lnot B\Rightarrow \lnot A $
                    \item $A\land \lnot B\Leftrightarrow A \Rightarrow B$
                    \item $C\Rightarrow A\lor \lnot B\Leftrightarrow \lnot A\land C $
                    \item $\lnot A\lor \lnot B\Leftrightarrow B \land (C\Leftrightarrow \lnot A) $
                \end{itemize}

        \begin{frame}
            \frametitle{Mno≈æice}
            \begin{alertblock}{Mno≈æica}
                \textbf{Mno≈æica} je skupek elementov, ki imajo neko skupno lastnost.

            \end{alertblock}
            \begin{block}{}
                Mno≈æice oznaƒçujemo z velikimi ƒçrkami ($\mathcal{A}, \mathcal{B}, \mathcal{C} \dots$ ali  $A, B, C \dots$ ). 
                
                Njeni elementi so zapisani v zavitem oklepaju (npr. $\mathcal{A}=\left\{2, 5, 7\right\}$).
            \end{block}

            \begin{alertblock}{Element mno≈æice}
            \textbf{Element mno≈æice} je objekt v mno≈æici. 
            \end{alertblock}
            \begin{block}{}
                Oznaƒçujemo jih z malimi ƒçrkami ($a, b, c \dots$). 

                Element mno≈æici lahko pripada ($a\in\mathcal{A}$) ali pa ne ($a\notin\mathcal{A}$).
            \end{block} 
        \end{frame}
