        \end{frame}


        \begin{frame}
            \frametitle{Lastnosti korenskih funkcij}

            \vskip-1em
            \begin{block}{Lastnosti korenskih funkcij z lihim korenskim eksponentom}
                    \begin{itemize}
                        \item $D_f=\mathbb{R}$
                        \item $Z_f=\mathbb{R}$
                        \item So naraščajoče za vse $x\in\mathbb{R}$.
                        \item Grafi potekajo skozi točke $T_0(0,0)$, $T_1(1,1)$ in $T_2(-1,-1)$.
                        \item So negativne za $x\in(-\infty,0)$ in pozitivne za $x\in(0,\infty)$.
                        \item So neomejene.
                        \item So lihe -- grafi so simetrični glede na koordinatno izhodišče.
                        \item So konveksne za $x\in(-\infty,0)$ in konkavne za $x\in(0,\infty)$.
                        \item Imajo ničlo pri $x=0$.
                        \item Tangenta na krivuljo v ničli je ordinatna os.
                    \end{itemize}
                \end{block}
        \end{frame}

            \begin{block}{Lastnosti korenskih funkcij s sodim korenskim eksponentom}
                    \begin{itemize}
                        \item $D_f=[0,\infty)$
                        \item $Z_f=[0,\infty)$
                        \item So naraščajoče za vse $x\in\mathbb{R}$.
                        \item Grafi potekajo skozi točki $T_0(0,0)$ in $T_1(1,1)$.
                        \item So pozitivne za vse $x\in(0,\infty)$.
                        \item So navzdol omejene z $y=0$ in navzgor neomejene.
                        \item So konkavne za $x\in(0,\infty)$.
                        \item Imajo ničlo pri $x=0$.
                        \item Tangenta na krivuljo v ničli je ordinatna os.
                    \end{itemize}
                \end{block}
        \end{frame}
